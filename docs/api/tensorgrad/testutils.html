 22def assert_close(actual, expected, rtol=1e-4, atol=1e-5):
 23    assert set(actual.names) == set(expected.names), f"{actual.names=} != {expected.names=}"
 24    actual = actual.align_to(*expected.names).rename(None)
 25    expected = expected.expand_as(actual).rename(None)
 26    torch.testing.assert_close(actual, expected, rtol=rtol, atol=atol)

111def make_random_tree(nodes: int):
112    components = [i for i in range(nodes)]
113
114    def find(x):
115        if components[x] != x:
116            components[x] = find(components[x])
117        return components[x]
118
119    def union(x, y):
120        components[find(x)] = find(y)
121
122    edges = []
123    adj = [[] for _ in range(nodes)]
124    while len(edges) < nodes - 1:
125        x, y = random.randint(0, nodes - 1), random.randint(0, nodes - 1)
126        if len(adj[x]) < 3 and len(adj[y]) < 3 and find(x) != find(y):
127            union(x, y)
128            edges.append((x, y))
129            adj[x].append(y)
130            adj[y].append(x)
131
132    # 3n edges, n-1 used, 2(n-1) used for connections, n+1 leaf nodes, 1 free edge left.
133    names = string.ascii_uppercase
134    vectors = []
135    variables = []
136    for i in range(nodes):
137        ts = [f"{names[min(i,j)]}|{names[max(i,j)]}" for j in adj[i]]
138        while len(ts) < 3 and len(vectors) < nodes + 1:
139            vi = len(vectors)
140            vectors.append(Variable(f"V{vi}", f"v{vi}"))
141            ts.append(f"v{vi}")
142        if len(ts) != 3:
143            ts.append("free")
144        variables.append(Variable(names[i], ts))
145
146    return vectors, variables

157def random_tensor_expr(max_depth=4, max_dim=4) -> tuple[Tensor, torch.Tensor, dict[Variable, torch.Tensor]]:
158    assert max_dim >= 1
159    if max_dim == 1:
160        symbols_list = [symbols("a")]
161    else:
162        symbols_list = symbols(" ".join(string.ascii_letters[:max_dim]))
163    sizes = {s: random.randrange(1, 2 * max_dim + 1) for s in symbols_list}
164    vars = [
165        (
166            Variable(f"var_{'_'.join(map(str,symbols))}", *symbols),
167            torch.randn([sizes[s] for s in symbols], names=list(map(str, symbols))),
168        )
169        for r in range(1, len(symbols_list) + 1)
170        for symbols in itertools.combinations(symbols_list, r)
171    ]
172
173    def inner(depth):
174        if depth == 0:
175            # return random.choice(random.choice([vars, copys]))
176            return random.choice(vars)
177        left, left_torch = inner(depth - 1)
178        right, right_torch = inner(depth - 1)
179        rand = random.random()
180        if rand < 0.3:
181            left_aligned, right_aligned = broadcast_tensors(left_torch, right_torch)
182            k = random.randint(-2, 4)
183            return left * right**k, left_aligned * right_aligned**k
184        if rand < 0.6:
185            left_aligned, right_aligned = broadcast_tensors(left_torch, right_torch)
186            return left + right, left_aligned + right_aligned
187        else:
188            contracted = left.edges & right.edges
189            rhs = "".join(e for e in left_torch.names + right_torch.names if e not in contracted)
190            eq = f"{''.join(left_torch.names)},{''.join(right_torch.names)}->{rhs}"
191            torch_result = torch.einsum(eq, left_torch.rename(None), right_torch.rename(None))
192            return left @ right, torch_result.rename(*rhs)
193
194    tensor, tensor_torch = inner(depth=max_depth)
195    return tensor, tensor_torch, {v: t for v, t in vars}
