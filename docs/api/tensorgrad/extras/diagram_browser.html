112function makeLayout(frame) {
113  const stored = frame.metadata && frame.metadata.layout;
114  if (!stored || !stored.nodes) {
115    return {nodes: {}, boxes: {}, nodeSizes: {}, nodePorts: {}, freePorts: {}, wireRoutes: {}};
116  }
117  return {
118    nodes: Object.fromEntries(Object.entries(stored.nodes || {}).map(([k, v]) => [k, {x: v[0], y: v[1]}])),
119    boxes: Object.fromEntries(Object.entries(stored.boxes || {}).map(([k, v]) => [k, {x: v[0], y: v[1], w: v[2], h: v[3]}])),

121    nodePorts: Object.fromEntries(Object.entries(stored.nodePorts || {}).map(([k, v]) => [k, [...v]])),

136function autoFit() {
137  const layout = layouts[frameIndex] || layouts[0];
138  if (!layout) return;
139  const boxes = Object.values(layout.boxes || {});
140  const nodeEntries = Object.entries(layout.nodes || {});
141  const rects = boxes.concat(nodeEntries.map(([id, p]) => {
142    const s = (layout.nodeSizes && layout.nodeSizes[id]) || {w: 38, h: 30};
143    return {x: p.x - s.w / 2, y: p.y - s.h / 2, w: s.w, h: s.h};
144  }));
145  if (!rects.length) return;
146  const minX = Math.min(...rects.map(r => r.x)) - 56;
147  const maxX = Math.max(...rects.map(r => r.x + r.w)) + 68;
148  const minY = Math.min(...rects.map(r => r.y)) - 44;
149  const maxY = Math.max(...rects.map(r => r.y + r.h)) + 38;
150  const w = maxX - minX, h = maxY - minY;
151  view.scale = Math.min(2.25, Math.max(0.7, Math.min((canvas.clientWidth - 80) / w, (canvas.clientHeight - 80) / h)));
152  view.x = -((minX + maxX) / 2) * view.scale;
153  view.y = -((minY + maxY) / 2) * view.scale;
154}

169function addMathLabelCandidates(tex, candidates, className, alpha = 1) {
170  if (alpha <= 0.001 || !tex) return;
171  const candidate = chooseLabelCandidate(tex, candidates, className);
172  const s = screenPoint(candidate.point);
173  const el = document.createElement("div");
174  el.className = `math-label ${className || ""}`;
175  el.style.left = `${s.x}px`;
176  el.style.top = `${s.y}px`;
177  el.style.opacity = String(alpha);
178  el.style.fontSize = `${Math.max(0.85, view.scale).toFixed(3)}em`;
179  const anchor = candidate.anchor || "center";
180  if (anchor === "left") el.style.transform = "translate(0, -50%)";
181  if (anchor === "right") el.style.transform = "translate(-100%, -50%)";
182  if (anchor === "top-left") el.style.transform = "translate(0, 0)";
183  if (anchor === "above") el.style.transform = "translate(-50%, -100%)";
184  if (anchor === "below") el.style.transform = "translate(-50%, 0)";
185  if (window.katex) {
186    window.katex.render(tex, el, {throwOnError: false, displayMode: false});
187  } else {
188    el.textContent = tex.replaceAll("\\", "");
189  }
190  labelLayer.appendChild(el);
191  placedLabelRects.push(expandRect(el.getBoundingClientRect(), 2));
192}

194function chooseLabelCandidate(tex, candidates, className) {
195  let best = candidates[0];
196  let bestScore = Infinity;
197  for (const candidate of candidates) {
198    const rect = estimateLabelRect(tex, className, candidate.point, candidate.anchor || "center");
199    const score = (candidate.penalty || 0) + labelCollisionPenalty(rect);
200    if (score < bestScore) {
201      best = candidate;
202      bestScore = score;
203    }
204  }
205  return best;
206}

208function estimateLabelRect(tex, className, point, anchor) {
209  const s = screenPoint(point);
210  const clean = String(tex).replace(/\\[a-zA-Z]+|[{}\\]/g, "");
211  const scale = Math.max(0.85, view.scale);
212  let multiplier = 1.08;
213  if ((className || "").includes("wire-label")) multiplier = 0.8;
214  if ((className || "").includes("node-label")) multiplier = 1.24;
215  if ((className || "").includes("coef-label")) multiplier = 1.85;
216  if ((className || "").includes("derivative-label")) multiplier = 1.2;
217  const font = 13 * scale * multiplier;
218  const w = Math.max(12, clean.length * font * 0.48 + 6);
219  const h = font * 1.25 + 4;
220  return anchoredRect(s.x, s.y, w, h, anchor || "center");
221}

223function anchoredRect(x, y, w, h, anchor) {
224  if (anchor === "left") return {left: x, top: y - h / 2, right: x + w, bottom: y + h / 2};
225  if (anchor === "right") return {left: x - w, top: y - h / 2, right: x, bottom: y + h / 2};
226  if (anchor === "top-left") return {left: x, top: y, right: x + w, bottom: y + h};
227  if (anchor === "above") return {left: x - w / 2, top: y - h, right: x + w / 2, bottom: y};
228  if (anchor === "below") return {left: x - w / 2, top: y, right: x + w / 2, bottom: y + h};
229  return {left: x - w / 2, top: y - h / 2, right: x + w / 2, bottom: y + h / 2};
230}

232function labelCollisionPenalty(rect) {
233  let score = 0;
234  for (const other of placedLabelRects) score += overlapArea(rect, other) * 6;
235  if (rect.left < 6) score += (6 - rect.left) * 3;
236  if (rect.top < 46) score += (46 - rect.top) * 3;
237  if (rect.right > canvas.clientWidth - 6) score += (rect.right - canvas.clientWidth + 6) * 3;
238  if (rect.bottom > canvas.clientHeight - 6) score += (rect.bottom - canvas.clientHeight + 6) * 3;
239  return score;
240}

242function overlapArea(a, b) {
243  const x = Math.max(0, Math.min(a.right, b.right) - Math.max(a.left, b.left));
244  const y = Math.max(0, Math.min(a.bottom, b.bottom) - Math.max(a.top, b.top));
245  return x * y;
246}

252function draw() {
253  ctx.clearRect(0, 0, canvas.clientWidth, canvas.clientHeight);
254  labelLayer.textContent = "";
255  placedLabelRects = [];
256  ctx.save();
257  ctx.translate(canvas.clientWidth / 2 + view.x, canvas.clientHeight / 2 + view.y);
258  ctx.scale(view.scale, view.scale);
259
260  if (transition) {
261    const raw = (performance.now() - transition.start) / transition.duration;
262    const t = Math.min(1, Math.max(0, ease(raw)));
263    drawTransition(transition.from, transition.to, t);
264    if (raw >= 1) {
265      frameIndex = transition.to;
266      transition = null;
267    } else {
268      requestAnimationFrame(draw);
269    }
270  } else {
271    const frame = frames[frameIndex];
272    const layout = layouts[frameIndex];
273    drawScene(frame, layout, () => 1);
274  }
275  ctx.restore();
276}

278function drawScene(frame, layout, alphaFor) {
279  reserveDiagramGeometry(layout);
280  for (const box of frame.boxes) drawBox(box, layout, alphaFor("box", box.id));
281  for (const wire of frame.wires) drawWire(wire, layout, alphaFor("wire", wire.id), frame);
282  for (const node of frame.nodes) drawNode(node, layout, alphaFor("node", node.id));
283}

285function reserveDiagramGeometry(layout) {
286  for (const [id, p] of Object.entries(layout.nodes || {})) {
287    const size = (layout.nodeSizes && layout.nodeSizes[id]) || {w: 34, h: 30};
288    const a = screenPoint({x: p.x - size.w / 2 - 4, y: p.y - size.h / 2 - 4});
289    const b = screenPoint({x: p.x + size.w / 2 + 4, y: p.y + size.h / 2 + 4});
290    placedLabelRects.push({
291      left: Math.min(a.x, b.x),
292      top: Math.min(a.y, b.y),
293      right: Math.max(a.x, b.x),
294      bottom: Math.max(a.y, b.y)
295    });
296  }
297}

299function drawTransition(fromIndex, toIndex, t) {
300  const baseTransition = transitions[Math.min(fromIndex, toIndex)] || {matches: {nodes: {}, wires: {}, boxes: {}}};
301  const matches = fromIndex < toIndex ? (baseTransition.matches || {}) : invertMatches(baseTransition.matches || {});
302  const tr = {matches};
303  const fromFrame = frames[fromIndex], toFrame = frames[toIndex];
304  const fromLayout = layouts[fromIndex], toLayout = layouts[toIndex];
305  const interp = interpolateLayout(fromLayout, toLayout, tr.matches || {}, t);
306  const matchedOld = {
307    node: new Set(Object.keys((tr.matches && tr.matches.nodes) || {})),
308    wire: new Set(Object.keys((tr.matches && tr.matches.wires) || {})),
309    box: new Set(Object.keys((tr.matches && tr.matches.boxes) || {}))
310  };
311  const matchedNew = {
312    node: new Set(Object.values((tr.matches && tr.matches.nodes) || {})),
313    wire: new Set(Object.values((tr.matches && tr.matches.wires) || {})),
314    box: new Set(Object.values((tr.matches && tr.matches.boxes) || {}))
315  };
316  drawScene(fromFrame, fromLayout, (kind, id) => matchedOld[kind].has(id) ? 0 : 1 - t);
317  drawScene(toFrame, interp, (kind, id) => matchedNew[kind].has(id) ? 1 : t);
318}

325    boxes: invert(matches.boxes)

329function interpolateLayout(fromLayout, toLayout, matches, t) {
330  const out = {
331    nodes: Object.fromEntries(Object.entries(toLayout.nodes).map(([id, p]) => [id, {...p}])),
332    boxes: Object.fromEntries(Object.entries(toLayout.boxes).map(([id, r]) => [id, {...r}])),
333    nodeSizes: Object.fromEntries(Object.entries(toLayout.nodeSizes || {}).map(([id, r]) => [id, {...r}])),
334    nodePorts: Object.fromEntries(Object.entries(toLayout.nodePorts || {}).map(([id, ports]) => [id, [...ports]])),
335    freePorts: Object.fromEntries(Object.entries(toLayout.freePorts || {}).map(([id, ports]) => [id, [...ports]])),
336    wireRoutes: Object.fromEntries(Object.entries(toLayout.wireRoutes || {}).map(([id, route]) => [id, route.map(p => ({...p}))])),
337    wireLabelCandidates: Object.fromEntries(Object.entries(toLayout.wireLabelCandidates || {}).map(([id, candidates]) => [id, candidates.map(c => ({...c, point: {...c.point}}))]))
338  };
339  for (const [oldId, newId] of Object.entries(matches.nodes || {})) {
340    const a = fromLayout.nodes[oldId], b = toLayout.nodes[newId];
341    if (a && b) out.nodes[newId] = {x: lerp(a.x, b.x, t), y: lerp(a.y, b.y, t)};
342  }
343  for (const [oldId, newId] of Object.entries(matches.boxes || {})) {
344    const a = fromLayout.boxes[oldId], b = toLayout.boxes[newId];
345    if (a && b) out.boxes[newId] = {
346      x: lerp(a.x, b.x, t),
347      y: lerp(a.y, b.y, t),
348      w: lerp(a.w, b.w, t),
349      h: lerp(a.h, b.h, t)
350    };
351  }
352  for (const [oldId, newId] of Object.entries(matches.wires || {})) {
353    const a = fromLayout.wireRoutes && fromLayout.wireRoutes[oldId];
354    const b = toLayout.wireRoutes && toLayout.wireRoutes[newId];
355    if (a && b && a.length === b.length) {
356      out.wireRoutes[newId] = b.map((p, i) => ({x: lerp(a[i].x, p.x, t), y: lerp(a[i].y, p.y, t)}));
357    }
358  }
359  for (const [oldId, newId] of Object.entries(matches.wires || {})) {
360    const a = fromLayout.wireLabelCandidates && fromLayout.wireLabelCandidates[oldId];
361    const b = toLayout.wireLabelCandidates && toLayout.wireLabelCandidates[newId];
362    if (a && b && a.length === b.length) {
363      out.wireLabelCandidates[newId] = b.map((candidate, i) => ({
364        ...candidate,
365        point: {
366          x: lerp(a[i].point.x, candidate.point.x, t),
367          y: lerp(a[i].point.y, candidate.point.y, t)
368        }
369      }));
370    }
371  }
372  return out;
373}

405function drawWire(wire, layout, alpha = 1, frame = null) {
406  if (wire.kind === "term_port") return drawTermStub(wire, layout, alpha);
407  if (wire.kind === "derivative_passthrough") return drawDerivativePassthrough(wire, layout, alpha);
408  if (wire.kind === "sum_port" || wire.kind.endsWith("_passthrough")) return;
409  withAlpha(alpha, () => {
410  const pts = routeForWire(wire, layout);
411  if (pts.length < 2) return;
412  applyEdgeStyle();
413  ctx.beginPath();
414  ctx.moveTo(pts[0].x, pts[0].y);
415  for (let i = 1; i < pts.length; i++) drawCurveSegment(pts[i - 1], pts[i], wire);
416  ctx.stroke();
417  drawBoundaryDots(wire, pts, frame);
418  const endpointLabels = visibleEndpointLabels(wire);
419  if (endpointLabels.length >= 2) {
420    for (const item of endpointLabels) {
421      const p = pts[item.index];
422      const neighbor = pts[item.index === 0 ? 1 : item.index - 1];
423      const candidates = isDerivativeBoundaryEndpoint(wire.endpoints[item.index], frame)
424        ? labelCandidatesForWire(layout, `${wire.id}:${item.index}`, derivativeBoundaryLabelCandidates(p, neighbor))
425        : labelCandidatesForWire(layout, `${wire.id}:${item.index}`, labelCandidatesNearEndpoint(p, neighbor));
426      addMathLabelCandidates(texForLabel(item.label), candidates, "wire-label", alpha);
427    }
428  } else if (wire.label) {
429    const derivativeIndex = wire.endpoints.findIndex(e => isDerivativeBoundaryEndpoint(e, frame));
430    if (derivativeIndex >= 0) {
431      const p = pts[derivativeIndex];
432      const neighbor = pts[derivativeIndex === 0 ? 1 : derivativeIndex - 1];
433      addMathLabelCandidates(texForLabel(wire.label), labelCandidatesForWire(layout, wire.id, derivativeBoundaryLabelCandidates(p, neighbor)), "wire-label", alpha);
434    } else {
435      addMathLabelCandidates(texForLabel(wire.label), labelCandidatesForWire(layout, wire.id, labelCandidatesOnWire(pts)), "wire-label", alpha);
436    }
437  }
438  });
439}

450function isDerivativeBoundaryEndpoint(endpoint, frame) {
451  if (!frame || endpoint.owner_kind !== "box") return false;
452  const box = frame.boxes.find(b => b.id === endpoint.owner);
453  return !!box && box.kind === "derivative";

456function derivativeBoundaryLabelCandidates(point, neighbor) {
457  const dx = point.x - neighbor.x, dy = point.y - neighbor.y;
458  const len = Math.hypot(dx, dy) || 1;
459  const along = {x: dx / len, y: dy / len};
460  const normal = {x: -along.y, y: along.x};
461  const base = {x: point.x + along.x * 18, y: point.y + along.y * 18};
462  const side = point.x <= neighbor.x ? 1 : -1;
463  return [
464    {point: {x: point.x + side * 17, y: point.y - 21}, anchor: "center", penalty: 0},
465    {point: {x: base.x + normal.x * 17, y: base.y + normal.y * 17 - 7}, anchor: "center", penalty: 12},
466    {point: {x: base.x - normal.x * 17, y: base.y - normal.y * 17 - 7}, anchor: "center", penalty: 24}
467  ];
468}

470function drawDerivativePassthrough(wire, layout, alpha = 1) {
471  const nodeEndpoint = wire.endpoints.find(e => e.owner_kind === "node" && layout.nodes[e.owner]);
472  const boxEndpoint = wire.endpoints.find(e => e.owner_kind === "box" && layout.boxes && layout.boxes[e.owner]);
473  if (!nodeEndpoint || !boxEndpoint) return;
474  withAlpha(alpha, () => {
475    const route = routeForWire(wire, layout);
476    if (route.length < 2) return;
477    const start = route[0];
478    const end = route[route.length - 1];
479    applyEdgeStyle();
480    ctx.beginPath();
481    ctx.moveTo(start.x, start.y);
482    ctx.lineTo(end.x, end.y);
483    ctx.stroke();
484    if (nodeEndpoint.label) {
485      addMathLabelCandidates(texForLabel(nodeEndpoint.label), labelCandidatesNearEndpoint(start, end, 0.65), "wire-label", alpha);
486    }
487  });
488}

543  const dx = b.x - a.x, dy = b.y - a.y;
544  const len = Math.hypot(dx, dy) || 1;

601function drawCurveSegment(a, b, wire) {
602  const dx = b.x - a.x, dy = b.y - a.y;
603  const dist = Math.hypot(dx, dy) || 1;
604  const multiplicity = (wire.metadata && wire.metadata.multiplicity) || 1;
605  if (wire.kind === "free" || multiplicity <= 1) {
606    ctx.lineTo(b.x, b.y);
607    return;
608  }
609  const nx = -dy / dist, ny = dx / dist;
610  const bend = Math.min(14, dist * 0.1) * (multiplicity % 2 ? 1 : -1);
611  const c1 = {x: a.x + dx * 0.42 + nx * bend, y: a.y + dy * 0.22 + ny * bend};
612  const c2 = {x: b.x - dx * 0.42 + nx * bend, y: b.y - dy * 0.22 + ny * bend};
613  ctx.bezierCurveTo(c1.x, c1.y, c2.x, c2.y, b.x, b.y);
614}

709function roundedRectPath(ctx, x, y, w, h, r) {
710  if (!r) {
711    ctx.beginPath();
712    ctx.rect(x, y, w, h);
713    return;
714  }

716  ctx.beginPath();
717  ctx.moveTo(x + rr, y);
718  ctx.lineTo(x + w - rr, y);
719  ctx.quadraticCurveTo(x + w, y, x + w, y + rr);
720  ctx.lineTo(x + w, y + h - rr);
721  ctx.quadraticCurveTo(x + w, y + h, x + w - rr, y + h);
722  ctx.lineTo(x + rr, y + h);
723  ctx.quadraticCurveTo(x, y + h, x, y + h - rr);
724  ctx.lineTo(x, y + rr);
725  ctx.quadraticCurveTo(x, y, x + rr, y);
726}

728function texForLabel(label) {
729  if (label == null) return "";
730  const text = String(label);
731  if (text === "+") return "+";
732  if (text.startsWith("+")) return "+\\," + texEscape(text.slice(1));
733  if (text === "-") return "-";
734  if (text.startsWith("d/d")) return "\\frac{d}{d" + texEscape(text.slice(3)) + "}";
735  if (text === "Delta") return "\\Delta";
736  return texEscape(text);
737}

742    .replaceAll("_", "'")
743    .replaceAll("{", "\\{")
744    .replaceAll("}", "\\}");
745}
