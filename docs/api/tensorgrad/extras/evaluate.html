 37    def __init__(
 38        self,
 39        values: dict[Variable, torch.Tensor],
 40        dims: dict[Symbol, int] | None = None,
 41    ):
 42        self.values = KeyStoringDict(values)
 43        self.dims = dims if dims is not None else {}
 44
 45        # Load sizes of variables into the dimensions dictionary and check consistency
 46        for v, t in values.items():
 47            if not isinstance(v, Variable):
 48                continue
 49            for e, ts in zip(t.names, t.shape):
 50                vs = v.shape[e]  # symbolic dimension
 51                if vs not in self.dims:
 52                    self.dims[vs] = ts
 53                elif self.dims[vs] != ts:
 54                    raise ValueError(f"Conflicting size for dim {e}")

 56    def evaluate(self, tensor: Tensor) -> torch.Tensor:
 57        # See if we have already evaluated a tensor isomorphic to this one
 58        if (other_tensor := self.values.get_with_key(tensor)) is not None:
 59            # Find the isomorphic representative that we matched
 60            other, torch_tensor = other_tensor
 61            assert torch_tensor.names == tuple(other.edges)
 62            mapping = next(other.isomorphisms(tensor), None)
 63            res = torch_tensor.rename(**mapping).align_to(*tensor.edges)
 64
 65            if __debug__:
 66                expected = self._evaluate(tensor)
 67                # We guarantee that inner_evaluate returns the edges in the same order as tensor.edges,
 68                # and res has had the order forced on it by align_to.
 69                assert expected.names == res.names, f"{expected.names=} {res.names=} {tensor.edges=}"
 70                torch.testing.assert_close(res.rename(None), expected.rename(None))
 71
 72            return res
 73
 74        res = self._evaluate(tensor)
 75
 76        if __debug__:
 77            assert not torch.isnan(res.rename(None)).any(), f"Got NaN in result in {tensor}"
 78            # We guarantee that inner_evaluate returns the edges in the same order as tensor.edges
 79            assert res.names == tuple(tensor.edges), f"Expected {tensor.edges=} but got {res.names=}"
 80            assert all(res.size(k) == self.dims[v] for k, v in tensor.shape.items() if v in self.dims)
 81
 82        # If the output defined a dimension size we didn't know about, store it
 83        # Note this is not a guaranteed metohd, since it creates a dependency on
 84        # the order of evaluation.
 85        for k, v in tensor.shape.items():
 86            if v not in self.dims:
 87                self.dims[v] = res.size(k)
 88
 89        self.values[tensor] = res
 90        return res

153    @_evaluate.register
154    def _(self, prod: Product):
155        # TODO: Keep track of how many contractions we made
156        # extras["contractions"] = extras.get("contractions", 0) + len(prod.contractions)
157        if not prod.factors:
158            return torch.tensor(1.0)
159        # We use "operator" einsum interface, which doesn't require single letter names.
160        # e.g.  einsum('i,i', b, b)  =  einsum(b, [0], b, [0])
161        # and   einsum('ij,jk->ik', b, c)  =  einsum(b, [0, 1], c, [1, 2], [0, 2])
162        edge_numbers = {e: i for i, e in enumerate({e for t in prod.factors for e in t.edges})}
163
164        # TODO: Merging copies is currently broken, because einsum doesn't allow us
165        # to use the same index twice in the output.
166        merge_copies = False
167        # We can't remove copies, if that's all we have
168        if all(isinstance(t, Delta) for t in prod.factors):
169            merge_copies = False
170        # We can't merge when there are repeated edges in the output, due to einsum
171        # limmitations
172        if len(set(edge_numbers[e] for e in prod.edges)) != len(prod.edges):
173            merge_copies = False
174
175        # We can make this more efficient by removing Delta tensors.
176        if merge_copies:
177            for t in prod.factors:
178                if isinstance(t, Delta):
179                    i0 = len(edge_numbers)
180                    for i, e in enumerate(t.edges):
181                        edge_numbers[e] = i0
182        parts = []
183        for t in prod.factors:
184            if not merge_copies or not isinstance(t, Delta):
185                torch_tensor = self.evaluate(t)
186                parts.append(torch_tensor.rename(None))
187                parts.append([edge_numbers[e] for e in torch_tensor.names])
188        parts.append([edge_numbers[e] for e in prod.edges])
189        return torch.einsum(*parts).rename(*prod.edges)

200    @_evaluate.register
201    def _(self, conv: Convolution):
202        w_in = self.dims.get(conv.shape[conv.input_name])
203        k_size = self.dims.get(conv.shape[conv.kernel_name])
204        w_out = self.dims.get(conv.shape[conv.output_name])
205
206        # We only need 2/3 of the input sizes to be given
207        if Counter([w_in, k_size, w_out])[None] >= 2:
208            raise ValueError(f"Convolution expects >= 2 of {conv.shape.keys()} to be given")
209        if w_in is None:
210            w_in = w_out + k_size - 1
211        elif k_size is None:
212            k_size = w_in - w_out + 1
213        elif w_out is None:
214            w_out = w_in - k_size + 1
215        elif w_out != w_in - k_size + 1:
216            raise ValueError(f"{w_out=} != {w_in=} - {k_size=} + 1")
217
218        # Make a tensor T, such that T[i,j,k] = 1 iff i=j+k
219        res = torch.zeros(w_in, k_size, w_out)
220        for k in range(w_out):
221            for j in range(k_size):
222                res[k + j, j, k] = 1
223        return res.rename(conv.input_name, conv.kernel_name, conv.output_name)

225    @_evaluate.register
226    def _(self, reshape: Reshape):
227        if not set(reshape.shape.values()).issubset(self.dims.keys()):
228            diff = reshape.shape.values() - self.dims.keys()
229            raise ValueError(f"Dims {diff} not supplied to Reshape")
230        sizes = [self.dims[s] for s in reshape.shape.values()]
231        full = math.prod(sizes)
232        half = int(math.sqrt(full))

234            raise ValueError(f"{sizes=} must multiply to a square number")
235        return torch.eye(half).reshape(*sizes).rename(*reshape.edges)

254@evaluate_function.register
255def _(func: _MatrixInverseFunction, x: torch.Tensor) -> torch.Tensor:
256    if not func.edges.issubset(x.names):
257        raise ValueError(f"Input {x.names} didn't have all edges {func.edges}")
258    d1, d2 = func.edges
259    # torch.inverse assumes matrix dimensions are at the end.
260    # We swap d1 and d2 for z1, so it's the edges with
261    # the same name that cancel, and not the opposite name.
262    z = x.align_to(..., d2, d1)
263    z = x.align_to(..., d1, d2).rename(None)
264    out_names = x.align_to(..., d2, d1).names
265    return torch.inverse(z).rename(*out_names).align_to(*x.names)

334@evaluate_function.register
335def _(func: _SoftmaxFunction, x: torch.Tensor) -> torch.Tensor:
336    (dims,) = func.inputs
337    sizes = [x.size(d) for d in dims]
338    names = [d for d in x.names if d not in dims]
339    other_sizes = [x.size(n) for n in names]
340    # Softmax doesn't support named dimensions, so we have to rename them to None.
341    # We move the affected dimensions to the front, then flatten them.
342    y = x.align_to(*dims, *names).rename(None).flatten(start_dim=0, end_dim=len(dims) - 1)
343    return y.softmax(dim=0).reshape(sizes + other_sizes).rename(*dims, *names).align_to(*x.names)
