 25def compile_to_callable(*tensors: Tensor, verbose=False, torch_compile=False):
 26    """
 27    Build a Python callable returning NumPy arrays:
 28        f(values, shapes=None) -> np.ndarray or tuple[np.ndarray, ...]
 29
 30    where:
 31      - `values` is a dict {Variable: np.ndarray}.
 32      - `shapes` is an optional dict {sympy.Symbol: int} with dimension sizes.
 33      - The code generator inspects all Symbols in the given Tensors, sorts them by
 34        symbol name, and uses them in the generated function signature.
 35
 36    The generated function `_generated_forward(...)` will be something like:
 37
 38        def _generated_forward(batch, c0, w0, h0, _var_1234, _var_5678, ...):
 39            ...
 40            return result
 41
 42    This function is then executed via `eval(...)`.
 43    """
 44    context = CodegenContext()
 45    final_var_names = [context._emit_tensor(t) for t in tensors]
 46    script = "\n".join(context.lines)
 47
 48    function_name = "_generated_forward"
 49
 50    # Sort the symbol names so we have a deterministic argument order
 51    all_symbols_sorted = sorted(context.seen_symbols, key=lambda s: s.name)
 52
 53    # Gather placeholders for each Variable
 54    var_placeholders = [f"_var_{id(v)}" for v in context.seen_variables]
 55
 56    # The final signature is: def _generated_forward(<dims>, <vars>):
 57    # e.g. def _generated_forward(batch, w0, out, _var_140661515588336, _var_140661515588464, ...):
 58    symbol_args = [f"{context.declare_dimension(s)}" for s in all_symbols_sorted]
 59    all_args = symbol_args + var_placeholders
 60    signature = ", ".join(all_args)
 61
 62    out_type = "np.ndarray" if len(tensors) == 1 else "tuple[np.ndarray, ...]"
 63
 64    # Build the generated function as a string
 65    wrapped_code = f"""\
 66def {function_name}({signature}) -> {out_type}:
 67{textwrap.indent(script, "    ")}
 68    return {', '.join(final_var_names)}
 69"""
 70    # Maybe later we can support numba
 71    # if torch_compile:
 72    #     wrapped_code = "@numba.jit()\n" + wrapped_code
 73
 74    # TODO: How can we move work that doesn't depend on the inputs out of the function?
 75
 76    if verbose:
 77        print()
 78        print("\n".join(f"{i+1:3}: {line}" for i, line in enumerate(wrapped_code.split("\n"))))
 79
 80    # Prepare a namespace for exec
 81    ns = {}
 82    exec(
 83        wrapped_code,
 84        {
 85            "np": np,
 86            "oe": oe,
 87            "math": math,
 88            "sparse": sparse,
 89        },
 90        ns,
 91    )
 92    generated_forward = ns[function_name]
 93
 94    # Keep track of how many times the inner function is called
 95    n_calls = 0
 96
 97    def forward_with_values(values: dict[Variable, torch.Tensor], shapes: dict[sympy.Symbol, int] = None):
 98        """
 99        Invoke the generated function `_generated_forward(...)` by passing
100        dimension sizes in sorted order (by symbol name), followed by the
101        variable arrays from `values`.
102        """
103        if shapes is None:
104            shapes = {}
105
106        # Populate shapes dict from the provided arrays
107        for v, arr in values.items():
108            if not isinstance(v, Variable):
109                continue
110            for dim_symbol, size in zip(v.shape.keys(), arr.shape):
111                # `dim_symbol` is a sympy.Symbol for that dimension
112                if dim_symbol not in shapes:
113                    shapes[dim_symbol] = size
114                elif shapes[dim_symbol] != size:
115                    raise ValueError(
116                        f"Conflicting size for symbol {dim_symbol}: "
117                        f"got both {shapes[dim_symbol]} and {size}"
118                    )
119
120        # Check we got the right dimension arguments
121        missing = set(all_symbols_sorted) - set(shapes.keys())
122        if missing:
123            raise ValueError(f"Missing dimension values for symbols: {missing}")
124
125        # Build the local namespace
126        local_ns = {}
127        local_ns[function_name] = generated_forward
128
129        # Insert dimension sizes
130        for sym in all_symbols_sorted:
131            local_ns[sym.name] = shapes[sym]
132
133        # Insert variable arrays
134        for var in context.seen_variables:
135            placeholder_name = f"_var_{id(var)}"
136            if var not in values:
137                raise KeyError(f"No value provided for variable {var.name}")
138            local_ns[placeholder_name] = values[var].rename(None).cpu().numpy()
139
140        # Evaluate the function call:
141        call_args = [s.name for s in all_symbols_sorted] + [f"_var_{id(v)}" for v in context.seen_variables]
142        call_expr = f"{function_name}({', '.join(call_args)})"
143
144        nonlocal n_calls
145        if verbose and n_calls == 0:
146            print(f"{call_expr}")
147        n_calls += 1
148
149        outputs = eval(call_expr, {}, local_ns)
150
151        # If there's only one output, wrap it in a tuple for uniformity
152        if len(tensors) == 1:
153            outputs = (outputs,)
154
155        # Wrap back in pytorch named tensors
156        wrapped = []
157        for numpy_array, original_tensor in zip(outputs, tensors):
158            if not isinstance(numpy_array, np.ndarray):
159                numpy_array = np.array(numpy_array)
160            wrapped.append(torch.from_numpy(numpy_array).refine_names(*original_tensor.edges))
161
162        # Return as a tuple or a single array
163        if len(wrapped) == 1:
164            return wrapped[0]
165        return tuple(wrapped)
166
167    return forward_with_values

186    def fresh_name(self, base: str):
187        """Generate a fresh variable name each time it is called."""
188        name = base
189        i = 1
190        while name in self.seen_names:
191            name = f"{base}_{i}"
192            i += 1
193        self.seen_names.add(name)
194        return name

196    def declare_dimension(self, symbol):
197        """
198        Return a string naming this dimension: either:
199          - str(symbol) if it's an integer
200          - symbol.name (with uniqueness checking) if it's a Sympy Symbol
201        """
202        if isinstance(symbol, int):
203            return str(symbol)
204        if isinstance(symbol, sympy.Symbol):
205            if symbol not in self.seen_symbols:
206                # Ensure we don't collide with e.g. loop variables
207                name = self.fresh_name(symbol.name)
208                self.seen_symbols[symbol] = name
209            return self.seen_symbols[symbol]
210        raise TypeError(f"Expected int or sympy.Symbol, got {symbol} (type={type(symbol)})")

212    def set_var_type(self, var_name, var_type):
213        self.var_types[var_name] = var_type

218    def _emit_tensor(self, tensor: Tensor) -> str:
219        """
220        Return the name of a local variable in the code that holds `tensor`.
221        If we've already generated code for an isomorphic version of `tensor`,
222        return the existing name; otherwise generate new code.
223        """
224        if (val := self.cache.get(tensor)) is not None:
225            other, name = val
226            # Attempt to see if the current `tensor` is a permutation of `other`
227            mapping = next(tensor.isomorphisms(other), None)
228            if mapping is not None:
229                # Build a permutation to reorder the existing name
230                other_edges = list(other.edges)
231                perm = [other_edges.index(mapping[e]) for e in tensor.edges]
232                if perm != list(range(tensor.order)):
233                    return f"{name}.transpose({perm})"
234                return name
235
236        var_name = self._emit_tensor_impl(tensor)
237        self.cache[tensor] = (tensor, var_name)
238        return var_name

407    @_emit_tensor_impl.register
408    def _(self, t: Function) -> str:
409        signature = t.signature
410        name = "".join(c for c in signature.name if c.isalnum())
411        var_name = self.fresh_name(f"fn_{name}")
412        child_names = [self._emit_tensor(inp) for inp in t.inputs]
413
414        if isinstance(signature, F._PowerFunction):
415            code_str = f"np.power({child_names[0]}, {signature.k})"
416            code_str += permute_code(t.inputs[0], t)
417        elif signature.name in ("exp", "log"):
418            code_str = f"np.{signature.name}({child_names[0]})"
419            code_str += permute_code(t.inputs[0], t)
420        elif signature.name == "relu":
421            code_str = f"np.maximum(0, {child_names[0]})"
422            code_str += permute_code(t.inputs[0], t)
423        elif signature.name == "gt0":
424            code_str = f"({child_names[0]} >= 0).astype(np.float32)"
425            code_str += permute_code(t.inputs[0], t)
426        elif signature.name == "argmax":
427            edges = list(t.inputs[0].edges)
428            dim = edges.index(signature.dim)
429            code_str = f"np.argmax({child_names[0]}, axis={dim})"
430            del edges[dim]
431            perm = [edges.index(e) for e in t.edges]
432            if perm != list(range(t.order)):
433                code_str += f".transpose({', '.join(map(str, perm))})"
434            assert not t.shape_out, "argmax should have no output dims"
435        else:
436            raise NotImplementedError(f"Unknown function {t.signature}")
437
438        self.emit(f"{var_name} = {code_str}  # edges: {', '.join(t.edges)}")
439        return var_name

456def permute_code(source: Tensor, target: Tensor) -> str:
457    """
458    Return a string for `.transpose(...)` in NumPy if needed, to reorder
459    'source' edges into 'target' edges.
460    """
461    source_edges = list(source.edges)
462    perm = [source_edges.index(e) for e in target.edges]
463    if perm == list(range(len(perm))):
464        return ""
465    return f".transpose({perm})"

153        if len(tensors) == 1:

157        wrapped = []
158        for numpy_array, original_tensor in zip(outputs, tensors):

161            wrapped.append(torch.from_numpy(numpy_array).refine_names(*original_tensor.edges))

164        if len(wrapped) == 1:

166        return tuple(wrapped)

220        """

222        If we've already generated code for an isomorphic version of `tensor`,
223        return the existing name; otherwise generate new code.
