  37def taylor(f: Tensor, wrt: Variable, eps: Tensor, n: int) -> Tensor:
  38    """Return the nth order Taylor approximation of f at x+eps."""
  39    if eps.shape != wrt.shape:
  40        raise ValueError("eps must have the same shape as wrt.")
  41    terms = [f]
  42    for _ in range(n):
  43        # Taking the derivative of f wrt wrt, creates new edges that match the shape of eps
  44        connection_names = _unused_edge_names(wrt.edges, f.edges)
  45        # To save time, we reuse the derivatives we've already computed
  46        f = f.grad(wrt, new_names=connection_names) @ eps.rename(**connection_names)
  47        terms.append(f)
  48    return Sum(terms, [Fraction(1, math.factorial(i)) for i in range(n + 1)])

  55def _is_even_permutation(permutation: Sequence[int]) -> bool:
  56    """
  57    Checks if given permutation is even.
  58    >>> is_even_permutation(range(10))
  59    True
  60    >>> is_even_permutation(range(10)[::-1])
  61    False
  62    """
  63    if len(permutation) == 1:
  64        return True
  65    transitions_count = 0
  66    for index, element in enumerate(permutation):
  67        for next_element in permutation[index + 1 :]:
  68            if element > next_element:
  69                transitions_count += 1
  70    return not (transitions_count % 2)

  76def symmetrize(t: Tensor, dims: DimType = None, signed: bool = False) -> Tensor:
  77    """Sum over all permutations of the edges."""
  78    dims = parse_dim(t.edges, dims, none_is="all")
  79    edges = list(dims)
  80    res: list[Tensor] = []
  81    weights: list[Number] = []
  82    for i_perm in itertools.permutations(range(len(edges))):
  83        perm = [edges[i] for i in i_perm]
  84        res.append(t.rename(**dict(zip(edges, perm))))
  85        if signed:
  86            weights.append(1 if _is_even_permutation(i_perm) else -1)
  87        else:
  88            weights.append(1)
  89    return Sum(res, weights)

  92def graph(dot_graph: str, **vars: Tensor) -> Tensor:
  93    """
  94    Create a tensor network using a DOT-like graph syntax.
  95
  96    This function allows you to define tensor networks using a syntax similar to the DOT graph description language.
  97    It provides a more intuitive way to specify tensor contractions and operations compared to traditional einsum notation.
  98
  99    Parameters:
 100    -----------
 101    dot_graph : str
 102        A string describing the tensor network using DOT-like syntax. Each line represents an edge or connection
 103        between tensors. Tensor names are separated by edge specifications, which are hyphen-delimited lists of
 104        edge names.
 105
 106    **vars : dict
 107        Keyword arguments representing the tensors used in the graph. Each key is the tensor name used in the
 108        dot_graph, and the corresponding value is the actual tensor object.
 109
 110    Returns:
 111    --------
 112    Product
 113        A Product object representing the tensor network described by the input graph.
 114
 115    Syntax:
 116    -------
 117    - Tensors are represented by their names (e.g., 'A', 'B', 'X').
 118    - Edge connections are represented by hyphens: '-'. For example, 'A -i- B' connects edge 'i' of A to edge 'i' of B.
 119    - You can connect edges with different names. For example, 'A -i-j- B' connects edge 'i' of A to edge 'j' of B.
 120    - Delta tensors are created automatically when a name starting with '*', like '*3' is used.
 121    - Lines can be separated by newlines or semicolons.
 122    - Edges not mentioned in the graph are broadcasted.
 123
 124    Examples:
 125    ---------
 126    1. Matrix multiplication:
 127    >>> i, j, k = symbols("i j k")
 128    >>> A = Variable("A", i, j)
 129    >>> B = Variable("B", j, k)
 130    >>> result = graph("A -j- B", A=A, B=B)
 131
 132    2. Trace
 133    >>> i = symbols("i")
 134    >>> X = Variable("X", i, j=i)
 135    >>> result = graph('''
 136    ...     X -i- X
 137    ...     X -j- X
 138    ... ''', X=X)
 139
 140    3. Element-wise multiplication:
 141    >>> i, j = symbols("i j")
 142    >>> X = Variable("X", i, j)
 143    >>> Y = Variable("Y", i, j)
 144    >>> result = graph('''
 145    ...     -i- *0 -i- X -j- *1 -j-
 146    ...         *0 -i- Y -j- *1
 147    ... ''', X=X, Y=Y)
 148
 149    4. Frobenius norm (squared)
 150    >>> i, j = symbols("i j")
 151    >>> X = Variable("X", i, j)
 152    >>> result = graph('''
 153    ...     X1 -i- X2
 154    ...     X1 -j- X2
 155    ... ''', X1=X, X2=X)
 156
 157    Raises:
 158    -------
 159    ValueError
 160        If the graph specification is invalid, such as using undefined variables,
 161        invalid edge names, or inconsistent hyperedge sizes.
 162    """
 163    vars = vars.copy()
 164
 165    # Parse the graph, converting long lines into a list of edges:
 166    # st. "A -i-j- B" -> ("A", "i", "j", "B")
 167    # and "A -i-" -> ("A", "i", "i", None)
 168    # and "A -i- B - C" -> [("A", "i", "i", "B"), ("B", "i", "i", "C")]
 169    # and "*1 - *2" -> ("*1", None, None, "*2")
 170    lines = re.split(r"[\n;]", dot_graph.strip())
 171    edges = []
 172    for line in lines:
 173        parts = line.split()
 174        last_var = None
 175        for i in range(len(parts)):
 176            if parts[i].startswith("-"):
 177                _, *edge_names, _ = parts[i].split("-")
 178                if len(edge_names) == 1:
 179                    edge_names = [edge_names[0]] * 2
 180                next_var = parts[i + 1] if i + 1 < len(parts) else None
 181                edges.append((last_var, *edge_names, next_var))
 182            else:
 183                last_var = parts[i]
 184
 185    # Look for edges, e, not mentioned in the graph, and create "X -e- " lines for them
 186    for v_name, v in vars.items():
 187        v_unused_edges = set(v.edges)
 188        for v0, e0, e1, v1 in edges:
 189            if v0 == v_name:
 190                v_unused_edges -= {e0}
 191            if v1 == v_name:
 192                v_unused_edges -= {e1}
 193        for e in v_unused_edges:
 194            edges.append((v_name, e, e, None))
 195
 196    # Keep track of the hyperedges and their sizes
 197    # Keys are *i for some i, values are lists of edge names/symbols
 198    hyperedges: dict[str | tuple[str, str], list[str]] = defaultdict(list)
 199    hypersizes: DisjointSets[Any, Symbol] = DisjointSets()
 200
 201    # Keep track of free edge names we can use
 202    used_edges: set[str] = {e for v in vars.values() for e in v.edges}
 203    free_edges: Iterator[str] = (f"e{i}" for i in itertools.count() if f"e{i}" not in used_edges)
 204    free_hyper_edges: Iterator[str] = (f"h{i}" for i in itertools.count() if f"h{i}" not in used_edges)
 205
 206    for v0, e0, e1, v1 in edges:
 207        # Syntax checking
 208        for v, e in ((v0, e0), (v1, e1)):
 209            if v and not v.startswith("*"):
 210                if v not in vars:
 211                    raise ValueError(f"Variable {v} not found in vars")
 212                if e not in vars[v].edges:
 213                    raise ValueError(f"Edge {e} not found in variable {v}({vars[v].edges})")
 214        if v0 is None and v1 is None:
 215            raise ValueError("Cannot have two free edges")
 216
 217        # The case 'X -i-' or '-i- X', where an edge is free
 218        elif v0 is None or v1 is None:
 219            v, e, eo = (v1, e1, e0) if v0 is None else (v0, e0, e1)
 220            # In the case "*0 -i-" we add the edge to the hyperedge
 221            if v.startswith("*"):
 222                hyperedges[v].append(e)
 223            # Otherwise we create a new hyperedge and add the edge to it
 224            # This allows us to keep track of broadcasting
 225            else:
 226                he = ("external", eo)
 227                c = next(free_edges)
 228                if eo not in hyperedges[he]:
 229                    hyperedges[he].append(eo)
 230                hyperedges[he].append(c)
 231                hypersizes[he] = vars[v].shape[e]
 232                vars[v] = vars[v].rename(**{e: c})
 233
 234        # The case *0 - *1, that is, two copy edges
 235        elif v0.startswith("*") and v1.startswith("*"):
 236            e = next(free_edges)
 237            hyperedges[v0].append(e)
 238            hyperedges[v1].append(e)
 239            hypersizes.union(v0, v1)  # Make sure they have the same size
 240
 241        # The case *0 -i- X, where we have a single copy edge
 242        elif v0.startswith("*") or v1.startswith("*"):
 243            v, e, he = (v0, e0, v1) if v1.startswith("*") else (v1, e1, v0)
 244            c = next(free_edges)
 245            hyperedges[he].append(c)
 246            hypersizes[he] = vars[v].shape[e]  # Fails if not compatible
 247            vars[v] = vars[v].rename(**{e: c})
 248
 249        # The case where a variable is connected to itself, like 'A -i- A'
 250        elif v0 == v1:
 251            if e0 == e1:
 252                raise ValueError("Cannot have a self loop on a single edge")
 253            c0, c1 = next(free_edges), next(free_edges)
 254            he = next(free_hyper_edges)
 255            hyperedges[he].extend([c0, c1])
 256            hypersizes[he] = vars[v0].shape[e0]
 257            vars[v0] = vars[v0].rename(**{e0: c0, e1: c1})
 258
 259        # A standard connection between two variables, like 'A -i- B'
 260        else:
 261            e = next(free_edges)
 262            vars[v0] = vars[v0].rename(**{e0: e})
 263            vars[v1] = vars[v1].rename(**{e1: e})
 264
 265    copies = []
 266    for he, edges in hyperedges.items():
 267        size = hypersizes.get(he)
 268        if size is None:
 269            raise ValueError(f"Hyperedge {he} has no size")
 270        if len(edges) != len(set(edges)):
 271            raise ValueError("Hyperedges must be disjoint")
 272        copies.append(Delta(size, *edges))
 273
 274    return Product(copies + list(vars.values()))

 286def diag(t: Tensor, new_edges: list[str]) -> Tensor:
 287    """If `t` is a vector, this creates a diagonal tensor with `t` and creates a diagonal.
 288    In einsum that means "i->iii".
 289    If `t` is a higher order tensor, with all dims the same size, this extracts the diagonal as a vector.
 290    In einsum that means "iii->i".
 291    """
 292    # Rename the edges to be distinct from the new edges
 293    (t,), _renames = _make_distinct(t, used_names=new_edges)
 294
 295    if not t.shape:
 296        # We can't just return t @ Delta(new_edges), since we don't know the size of the new edges.
 297        raise ValueError("Cannot take the diagonal of a scalar.")
 298
 299    edges, sizes = zip(*t.shape.items())
 300    if len(set(sizes)) != 1:
 301        raise ValueError("All dimensions must be the same size for the diagonal.")
 302
 303    return t @ Delta(sizes[0], *edges, *new_edges)

 312def parse_dim(tensor_edges: set[str], dim: DimType = None, none_is: str = "error") -> set[str]:

 314        if none_is == "all":
 315            dim = frozenset(tensor_edges)

 318    if isinstance(dim, str):
 319        dim = {dim}

 322    if not dim.issubset(tensor_edges):
 323        raise ValueError(f"{dim=} must be a subset of {tensor_edges}")
 324    return dim

 327def sum(tensor: Tensor, dim: DimType = None, keepdims: bool = False) -> Tensor:
 328    """Sum the tensor over the given dimensions."""
 329    dim = parse_dim(tensor.edges, dim, none_is="all")
 330    out = Product([tensor] + [Delta(tensor.shape[e], e) for e in dim])
 331    # Optionally broadcast back to orignal shape
 332    if keepdims:
 333        return out @ Ones(**{e: tensor.shape[e] for e in dim})
 334    return out

 337def _make_distinct(
 338    *tensors: Tensor, used_names: Optional[Iterable[str]] = None
 339) -> tuple[list[Tensor], list[dict[str, str]]]:
 340    """
 341    Ensure that the tensors have distinct edge names.
 342
 343    Args:
 344        *tensors: A sequence of tensors.
 345        used_names: Additional names that must be avoided.
 346
 347    Returns:
 348        A tuple containing:
 349        - List of tensors with renamed edges
 350        - List of rename mappings used for each tensor
 351    """
 352    # Delta the set, so we don't modify the input
 353    used_names = set() if used_names is None else set(used_names)
 354    # No reason to rename the names that are unique
 355    cnt = Counter(e for t in tensors for e in t.edges)
 356    unique_names = {e for e, c in cnt.items() if c == 1}
 357    # Unless the names are already used, then we can't save them
 358    unique_names -= used_names
 359    used_names |= unique_names
 360
 361    res, renames = [], []
 362    for i, t in enumerate(tensors):
 363        edges = t.edges - unique_names
 364        rename = _unused_edge_names(edges, used_names, suffix=f"_{i}")
 365        # Make sure all t.edges are present in rename
 366        rename.update({e: e for e in unique_names})
 367        used_names.update(rename.values())
 368        res.append(t.rename(**rename))
 369        renames.append(rename)
 370    return res, renames

 373def prod(*tensors: Tensor) -> Tensor:
 374    """Element-wise product of tensors."""
 375    shared_edges = set.intersection(*[set(t.edges) for t in tensors])
 376    for e in shared_edges:
 377        if not all(t.shape[e] == tensors[0].shape[e] for t in tensors):
 378            raise ValueError(f"Shapes of tensors don't match for edge {e}")
 379
 380    # Rename all edges to be distinct, and use names that won't be in the final output
 381    # Also doesn't rename unique names, which is good, since those will also be in the output
 382    ts, renames = _make_distinct(*tensors, used_names=shared_edges)
 383
 384    # Element-wise (Hadamard) product is easy to implement using Delta tensors
 385    ts += [Delta(tensors[0].shape[e], e, *(rn[e] for rn in renames)) for e in shared_edges]
 386
 387    res = Product(ts)
 388    assert res.shape == {e: s for t in tensors for e, s in t.shape.items()}
 389    return res

 392def mean(tensor: Tensor, dim: DimType = None, keepdims: bool = False) -> Tensor:
 393    dim = parse_dim(tensor.edges, dim, none_is="all")
 394    s = sum(tensor, dim, keepdims)
 395    normalization = 1
 396    for e in dim:
 397        normalization @= Delta(tensor.shape[e])
 398    return s / normalization

 401def dot(t1: Tensor, t2: Tensor, dim: str | tuple[str, str]) -> Tensor:
 402    """Contract two tensors along the given dimensions, broadcasting over the remaining shared edges.
 403    If the dimension is a tuple of two strings, the first string is the dimension of t1
 404    and the second is the dimension of t2."""
 405    # Like https://pytorch.org/docs/stable/generated/torch.tensordot.html
 406    # TODO: Should support a list of strings, or a tuple[list[str], list[str]]
 407    if isinstance(dim, str):
 408        dim = (dim, dim)
 409    if len(dim) != 2 or not all(isinstance(d, str) for d in dim):
 410        raise ValueError(f"Dot product requires one or two dimensions, got {dim=}")
 411    if dim[0] not in t1.edges or dim[1] not in t2.edges:
 412        raise ValueError(f"Edges {dim} must be in the tensors")
 413    free_name = dim[0] + "_"
 414    while free_name in t1.edges | t2.edges:
 415        free_name += "_"
 416    prod = t1.rename(**{dim[0]: free_name}) * t2.rename(**{dim[1]: free_name})
 417    return sum(prod, free_name)

 420def multi_dot(ts: list[Tensor], dims: tuple[str, str]) -> Tensor:
 421    """
 422    Compute the dot product of two or more tensors.
 423    """
 424    if not ts:
 425        return Ones()
 426    if any(v for v in ts if not isinstance(v, Tensor)):
 427        raise ValueError(f"All arguments must be tensors, got {ts}")
 428    prod = ts[0]
 429    assert isinstance(prod, Tensor)
 430    for t2 in ts[1:]:
 431        # We use the right edge of t and the left edge of t2
 432        prod = dot(prod, t2, dim=(dims[1], dims[0]))
 433        assert isinstance(prod, Tensor)
 434    return prod

 465class _PowerFunction(FunctionSignature):
 466    def __init__(self, k: int):
 467        self.name = f"pow({k=})"
 468        self.edges = frozenset()
 469        self.inputs = (frozenset(),)
 470        self.k = k
 471
 472    def derivative(self, i: int, new_edges: dict[str, str] = None) -> FunctionSignature:
 473        assert i == 0 and (new_edges is None or not new_edges)
 474        return _ScaleFunction(_PowerFunction(self.k - 1), self.k)
 475
 476    def simplify(self, func: Function, args: dict[str, Any]) -> Tensor:
 477        assert func.signature is self
 478        assert len(func.inputs) == 1, "pow should only have one input"
 479        (inner,) = func.inputs
 480
 481        if self.k == 0:
 482            return Ones(**func.shape)
 483        if self.k == 1:
 484            return inner.simplify(args)
 485
 486        # The pow function is multiplicative, so we can pull components out of a product apart.
 487        # Ie. (A * B)^k = A^k * B^k
 488        if isinstance(inner, Product):
 489            assert func.shape_out == {}
 490            new_comps = [Function(self, (comp,), {}) for comp in inner.components()]
 491            # We only apply this if we actually got more than one component
 492            if len(new_comps) > 1:
 493                return Product(new_comps).simplify(args)
 494
 495        # We can pull out the weight of a sum if it's just a single tensor
 496        if isinstance(inner, Sum) and len(inner.terms) == 1:
 497            (w,) = inner.weights
 498            (t,) = inner.terms
 499            return Function(self, (t,), {}) * (w**self.k)
 500
 501        # Base cases. We could add Copy(d)**k = Copy(d**k) here, but we're trying to keep the
 502        # mathematical expressions in tensorgrad space, rather than in sympy space.
 503        if isinstance(inner, Zero) or isinstance(inner, Product) and not inner.factors:
 504            return inner
 505
 506        # Combine nested pows
 507        if isinstance(inner, Function) and isinstance(inner.signature, _PowerFunction):
 508            return Function(_PowerFunction(k=inner.signature.k * self.k), inner.inputs, func.shape_out)
 509
 510        return func

 512    @classmethod
 513    def simplify_outer(cls, tensors: list[Tensor], args: dict[str, Any] = None) -> list[Tensor]:
 514        """Simplify a product by combining pow functions."""
 515        # This is not a general FunctionSignature method, but a special case for pow.
 516        original_edges = Product(tensors).edges
 517
 518        # New plan:
 519        #  - First try to combine existing pow functions.
 520        #  - Then try to do cancelations with non-pow subgraphs
 521        #  - Finally we could try to recreate some pow functions from the remaining subgraphs,
 522        #    but it's not clear that this is actually useful.
 523
 524        # Combine pow(x, k) * pow(x, l) = pow(x, k + l)
 525        tensors = cls._combine_powers(tensors)
 526        assert Product(tensors).edges == original_edges
 527
 528        if args["factor_components"]:
 529            # Combine  pow(x, k) * x = pow(x, k + 1)  or cancellations  pow(x, -1) * x = 1
 530            tensors = cls._combine_components(tensors)
 531            assert Product(tensors).edges == original_edges
 532
 533        # We have to merge products here because we might otherwise have undone part of the simplification
 534        tensors = Product.merge([Product([t]) if not isinstance(t, Product) else t for t in tensors]).factors
 535
 536        assert Product(tensors).edges == original_edges
 537        return tensors

 539    @classmethod
 540    def _combine_powers(cls, tensors: list[Tensor]) -> list[Tensor]:
 541        # Find an existing power, like pow(X, k=2), and merge other powers of X,
 542        # or instances of X, into it.
 543        seen = set()
 544        while True:
 545            # Find the next power function we haven't seen yet
 546            try:
 547                i, t = next(
 548                    (i, t)
 549                    for i, t in enumerate(tensors)
 550                    if isinstance(t, Function) and isinstance(t.signature, _PowerFunction) and t not in seen
 551                )
 552            except StopIteration:
 553                break
 554            seen.add(t)
 555            if t.signature.k > 5:
 556                break
 557
 558            power = t.signature.k
 559            (inner,) = t.inputs
 560
 561            hyperedges = {
 562                e: min(c.edges)
 563                for c in tensors
 564                if isinstance(c, Delta) and c.edges & inner.edges
 565                for e in c.edges
 566            }
 567            partition = defaultdict(list)
 568
 569            # We can't just call t.rename(**hyperedges) here, since some tensors might be adjacent to the
 570            # same hyperedge multiple times. Instead we give MatchEdgesKey the names, and it will perform
 571            # the isomorphism check correctly.
 572            partition[_MatchEdgesKey(inner, **hyperedges)].append((power, inner))
 573
 574            # We remove the tensor, as well as all Delta that it's connected to,
 575            # which makes the rest of the graph fall apart.
 576            others = tensors[:i] + tensors[i + 1 :]
 577            copys = [t for t in others if isinstance(t, Delta) and t.edges & inner.edges]
 578            others = [t for t in others if not (isinstance(t, Delta) and t.edges & inner.edges)]
 579            for comp in Product(others).components():
 580                power = 1
 581                if len(comp.factors) == 1:
 582                    comp = comp.factors[0]
 583                    if isinstance(comp, Function) and isinstance(comp.signature, _PowerFunction):
 584                        power = comp.signature.k
 585                        (comp,) = comp.inputs
 586                partition[_MatchEdgesKey(comp, **hyperedges)].append((power, comp))
 587
 588            # Now we have a partition of tensors that share the same edges, and we can combine them.
 589            # Or in some cases they cancel each other.
 590            tensors = copys
 591            for ts in partition.values():
 592                k = _sum(k for k, t in ts)
 593                (_, t0) = ts[0]  # All the tensors should be the same t
 594                tensors.append(pow(t0, k))
 595                # The remaining tensors have been reduced to ones. This prevents leaving unattached edges
 596                # on the copy tensors.
 597                for _, t in ts[1:]:
 598                    # Take care to not add a lot of empty Ones/Products, as it could
 599                    # make our method loop forever as it collects powers of nothing.
 600                    if t.shape:
 601                        tensors.append(Ones(**t.shape))
 602        return tensors
 603
 604    @classmethod
 605    def _combine_components(cls, tensors: list[Tensor]) -> list[Tensor]:
 606        # Look for tensors that share the same edges, and combine them.
 607        partition = defaultdict(int)
 608        for p in Product(tensors).components():
 609            power = 1
 610            if len(p.factors) == 1:
 611                (t,) = p.factors
 612                if isinstance(t, Function) and isinstance(t.signature, _PowerFunction):
 613                    power = t.signature.k
 614                    (t,) = t.inputs
 615            else:
 616                t = p
 617            partition[_MatchEdgesKey(t)] += power
 618
 619        tensors = []
 620        for key, w in partition.items():
 621            t = key.value
 622            tensors.append(pow(t, w))
 623        return tensors

 631        return tensor

 681class _MatrixInverseDerivative(FunctionSignature):
 682    def __init__(self, edges: set[str], new_edges: dict[str, str]) -> None:
 683        # Derivative gets new_edges : {old_name : new_name}
 684        super().__init__("inv_grad", edges | set(new_edges.values()), (edges,))
 685        self.new_edges = new_edges.copy()

 688        raise NotImplementedError("Please expand with simplify() first")

 690    def simplify(self, t: Function, args: dict[str, Any]):
 691        (edges,) = self.inputs
 692        (inner,) = t.inputs
 693        # Inverse: -inv(A)^T dA inv(A)
 694        if args["expand_functions"]:
 695            inv = inverse(inner, edges)
 696            #    -i- A^{-1} -j-j'-
 697            # -i'-i- A^{-1} -j-
 698            (o1, e1), (o2, e2) = self.new_edges.items()

 700            assert res.edges == t.edges

 705def inverse(tensor: Tensor, dims: set[str] = None) -> Tensor:
 706    """Matrix inverse over two target edges. Broadcasts over the rest.
 707    Mirrors torch.inverse / torch.linalg.inv.
 708
 709    In principle we could generalize the inverse to take a number of input
 710    edges, and the output would be a tensor with the same edges as the
 711    original, but if you contract it with the original over the selected
 712    edges, they cancel, and you get a tensor product over identitiy matrices
 713    over the remaining edges.
 714
 715    If dims is None, we assume the input tensor is a matrix.
 716    """
 717    if dims is None:
 718        dims = tensor.edges
 719    if len(dims) != 2:
 720        raise ValueError(f"Called Matrix Inverse with {dims=}")
 721    out_shape = {name: size for name, size in tensor.shape.items() if name in dims}
 722    s1, s2 = out_shape.values()
 723    if s1 != s2:
 724        raise ValueError(f"Inverted dimensions must have same size. Got {s1, s2}")
 725    return Function(_MatrixInverseFunction(dims), [tensor], out_shape)

 823        # with the same name, inputs and edges are considered equal.

 961def max_grad(t: Tensor, dim: DimType = None) -> Tensor:
 962    """
 963    Tie-splitting subgradient for max. If multiple elements tie for max,
 964    each gets 1/k of the gradient.
 965    Maps an order n tensor to an order n tensor.
 966
 967    Examples:
 968    (1)   j
 969        [1 4] -> max({i}) -> [2 4] -> grad({i}) -> [0 1] (broadcasted over j)
 970      i [2 3]                                      [1 0]
 971
 972    (2)   j
 973        [1 4] -> max({i,j}) -> [4] -> grad({i,j}) -> [0 1]
 974      i [2 3]                                        [0 0]
 975
 976    (3)   j
 977        [1 4] -> max({j}) -> [4] -> grad({j}) -> [0   1  ] (broadcasted over i)
 978      i [3 3]                [3]                 [0.5 0.5]
 979    """
 980    if isinstance(dim, str):
 981        dim = {dim}
 982    # Just as torch.amax, we default to "all dims".
 983    if dim is None:
 984        dim = set(t.edges)
 985
 986    # All the edges that are consumed by the function are also produced by the function
 987    func = Function(_MaxGradFunction(dim), [t], {e: t.shape[e] for e in dim})
 988    assert func.edges == t.edges
 989    return func

170    # and "*1 - *2" -> ("*1", None, None, "*2")
171    lines = re.split(r"[\n;]", dot_graph.strip())
172    edges = []
173    for line in lines:
174        parts = line.split()

176        for i in range(len(parts)):
177            if parts[i].startswith("-"):
178                _, *edge_names, _ = parts[i].split("-")
179                if len(edge_names) == 1:
180                    edge_names = [edge_names[0]] * 2
181                next_var = parts[i + 1] if i + 1 < len(parts) else None

183            else:

185

190            if v0 == v_name:

192            if v1 == v_name:

194        for e in v_unused_edges:

196

201

204    free_edges: Iterator[str] = (f"e{i}" for i in itertools.count() if f"e{i}" not in used_edges)
205    free_hyper_edges: Iterator[str] = (f"h{i}" for i in itertools.count() if f"h{i}" not in used_edges)
206
207    for v0, e0, e1, v1 in edges:
208        # Syntax checking
209        for v, e in ((v0, e0), (v1, e1)):
210            if v and not v.startswith("*"):
211                if v not in vars:

213                if e not in vars[v].edges:
214                    raise ValueError(f"Edge {e} not found in variable {v}({vars[v].edges})")
215        if v0 is None and v1 is None:
216            raise ValueError("Cannot have two free edges")
217

219        elif v0 is None or v1 is None:
220            v, e, eo = (v1, e1, e0) if v0 is None else (v0, e0, e1)
221            # In the case "*0 -i-" we add the edge to the hyperedge
222            if v.startswith("*"):
223                hyperedges[v].append(e)
224            # Otherwise we create a new hyperedge and add the edge to it
225            # This allows us to keep track of broadcasting
226            else:

231                hyperedges[he].append(c)
232                hypersizes[he] = vars[v].shape[e]

234
235        # The case *0 - *1, that is, two copy edges
236        elif v0.startswith("*") and v1.startswith("*"):
237            e = next(free_edges)
238            hyperedges[v0].append(e)
239            hyperedges[v1].append(e)
240            hypersizes.union(v0, v1)  # Make sure they have the same size
241
242        # The case *0 -i- X, where we have a single copy edge
243        elif v0.startswith("*") or v1.startswith("*"):
244            v, e, he = (v0, e0, v1) if v1.startswith("*") else (v1, e1, v0)
245            c = next(free_edges)
246            hyperedges[he].append(c)
247            hypersizes[he] = vars[v].shape[e]  # Fails if not compatible
248            vars[v] = vars[v].rename(**{e: c})
249
250        # The case where a variable is connected to itself, like 'A -i- A'
251        elif v0 == v1:
252            if e0 == e1:
253                raise ValueError("Cannot have a self loop on a single edge")
254            c0, c1 = next(free_edges), next(free_edges)
255            he = next(free_hyper_edges)
256            hyperedges[he].extend([c0, c1])
257            hypersizes[he] = vars[v0].shape[e0]
258            vars[v0] = vars[v0].rename(**{e0: c0, e1: c1})
259
260        # A standard connection between two variables, like 'A -i- B'
261        else:
262            e = next(free_edges)
263            vars[v0] = vars[v0].rename(**{e0: e})
264            vars[v1] = vars[v1].rename(**{e1: e})
265
266    copies = []
267    for he, edges in hyperedges.items():
268        size = hypersizes.get(he)
269        if size is None:
270            raise ValueError(f"Hyperedge {he} has no size")
271        if len(edges) != len(set(edges)):
272            raise ValueError("Hyperedges must be disjoint")
273        copies.append(Delta(size, *edges))
274
275    return Product(copies + list(vars.values()))
