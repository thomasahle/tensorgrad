
\chapter{Statistics and Probability}
\section{Definition of Moments}

Let $x\in\mathbb R^{n}$ be a random variable.
We write $m = E[x]\in\mathbb R^n$ for the expectation
and write $\tilde x = x - m$ for the centered variable,
so that $M=\mathrm{Var}[x] = E[\tilde x \tilde x^T]$ is the covariance (when these quantities are defined.)

In tensor diagrams, we will use square brackets:
\[
   \vecmatvec{1em}{}{}{m}
   =
\mathbin{\begin{tikzpicture}[baseline=(n0.base), inner sep=1pt]
   \node (n0) at (0,0) {$[$};
   \node [right=1em of n0] (n1) {$x]$};
   \draw (n0) -- (n1);
\end{tikzpicture}}
\quad\text{and}\quad
   \vecmatvec{1em}{}{M}{}
   =
\mathbin{\begin{tikzpicture}[baseline=(n0.base), inner sep=1pt]
   \node (n0) at (0,0) {$[$};
   \node [right=1em of n0] (n1) {$\tilde x$};
   \node [right=.5em of n1] (n2) {$\tilde x$};
   \node [right=1em of n2] (n3) {$]$};
   \draw (n0) -- (n1);
   \draw (n2) -- (n3);
\end{tikzpicture}}
\]
Inside diagrams we always write the centered variable as $\tilde x$ rather than $x - m$, since a bare minus sign inside a diagram could be mistaken for a contraction edge.

We can also define the third and fourth centralized moment tensors
\[
   \vcenter{\hbox{\import{figures/}{moment_tensors.pdf_tex}}}
\]
equal to the natural outer products
\[
   M_3 =
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{1em}{\tilde x}{}{} \\
      \vecmatvec{1em}{\tilde x}{}{} \\
      \vecmatvec{1em}{\tilde x}{}{}
   \end{bmatrix}
\quad\text{and}\quad
   M_4 =
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{1em}{\tilde x}{}{} \\
      \vecmatvec{1em}{\tilde x}{}{} \\
      \vecmatvec{1em}{\tilde x}{}{} \\
      \vecmatvec{1em}{\tilde x}{}{}
   \end{bmatrix}
.
\]

\subsection{Expectation of Linear Combinations}
General principle: The ``linearity of expectation'' lets you pull out all parts of the graph not involving $X$, which we assume here has mean zero, so that its third moment tensor is exactly the $M_3$ defined above.
\[
   \vcenter{\hbox{
      \import{figures/}{linearityOfExpectation.pdf_tex}
   }}
\]
The remaining tensor on the right (labeled $M_3$ in the figure) is an order-9 tensor that depends only on the distribution of $X$, not on the constants $A$, $B$, $C$, $D$, $E$.
In practice you would want to name the edges to keep track of what gets multiplied with what.

We can even use linearity of expectation to push the expectation inside an infinite sum of tensors, as in the following moment generating function, which relates all the $M_k$ tensors:
\begin{align*}
   \mathrm{E}(\mathrm{e}^{\langle \tilde x, t\rangle})
   &= \sum_{k=0}^\infty \frac{1}{k!} \mathrm{E}[\langle \tilde x, t\rangle^k]
    = \mathrm{E}\big[\sum_{k=0}^\infty \frac{1}{k!} \langle \tilde x^{\otimes k}, t^{\otimes k}\rangle\big]
   = \sum_{k=0}^\infty \frac{1}{k!} \langle M_k, t^{\otimes k}\rangle
 \\&=
  % =
   1
   +
   \frac{1}{2}
   \begin{tikzpicture}[baseline=(T.base), inner sep=1]
      \node (T) {$M$};
      \draw (T) -- ++(0,.5) node[fill=white] {$t$};
      \draw (T) -- ++(0,-.5) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \frac{1}{6}
   \begin{tikzpicture}[baseline=(T.base), inner sep=1]
      \node (T) {$M_3$};
      \draw (T) -- ++(0,.5) node[fill=white] {$t$};
      \draw (T) -- ++(-.4,-.4) node[fill=white] {$t$};
      \draw (T) -- ++(.4,-.4) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \frac{1}{24}
   \begin{tikzpicture}[baseline=(T.base), inner sep=1]
      \node (T) {$M_4$};
      \draw (T) -- ++(-.4,-.4) node[fill=white] {$t$};
      \draw (T) -- ++(.4,-.4) node[fill=white] {$t$};
      \draw (T) -- ++(-.4,.4) node[fill=white] {$t$};
      \draw (T) -- ++(.4,.4) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \dots
\end{align*}
Note that there is no linear term, since $M_1 = \E[\tilde x] = 0$.

\subsection{Linear Forms}
The Matrix Cookbook gives the following simple expectation:
\begin{walign}
   \tag{312}
   \E[AXB+C] &= A \E[X] B + C
   &
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \matmul{A,X,B} \\+\, \matmul{C}
   \end{bmatrix}
   &=
   \renewcommand*{\arraystretch}{1.3}
   \begin{matrix}
      \matmul{A,[X],B} \\+\, \matmul{C}
   \end{matrix}
\end{walign}

\subsection{Quadratic Forms}
We often prefer to write expectations in terms of the simple centered moments, which we can do by pulling out the mean:
\renewcommand*{\arraystretch}{1}
\[
   \begin{bmatrix}
      x - \\
      x -
   \end{bmatrix}
   =
   \begin{bmatrix}
      \tilde x - \\
      \tilde x -
   \end{bmatrix}
   +
   \begin{array}{c}
      m - \\
      m -
   \end{array}
   =
   \begin{tikzpicture}[baseline=(T.base), inner sep=1pt]
      \node (T) {$M$};
      \draw (T) -- ++(.5,.25);
      \draw (T) -- ++(.5,-.25);
   \end{tikzpicture}
   +
   \begin{array}{c}
      m - \\
      m -
   \end{array}
\]
This makes it easy to handle the quadratic forms from the Matrix Cookbook:
\begin{walign}
   \tag{313}
   \mathrm{Var}[Ax] &= A \mathrm{Var}[x] A^T
   &
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \begin{tikzpicture}[baseline=-.25em, inner sep=1pt]
         \draw (0,0) node[right] {$\widetilde{Ax}$} -- ++(-.3,0);
      \end{tikzpicture} \\
      \begin{tikzpicture}[baseline=-.25em, inner sep=1pt]
         \draw (0,0) node[right] {$\widetilde{Ax}$} -- ++(-.3,0);
      \end{tikzpicture}
   \end{bmatrix}
   &=
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{\tilde x} \\
      \vecmatvec{.5em}{A}{}{\tilde x}
   \end{bmatrix}
 \\&&&=
   \vcenter{\hbox{\begin{tikzpicture}[inner sep=1pt]
      \node (n1) at (0,-.25) {$\vecmatvec{1em}{A}{}{\tilde x}$};
      \node (n2) at (0,.25) {$\vecmatvec{1em}{A}{}{\tilde x}$};
      \node at (-.45, 0) {$\Bigg[$};
      \node at (.8, 0) {$\Bigg]$};
   \end{tikzpicture}}}
 \\&&& =
   \vecmatvec{.5em}{}{A,M_2,A}{}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
   \\[.5em]
   \tag{318}
   \E[x^T A x]
   &= \mathrm{Tr}(A M) + m^T A m
   &
   [\vecmatvec{.5em}{x}{A}{x}]
   &=
   \left(
      \begin{bmatrix}
         \tilde x - \\
         \tilde x -
      \end{bmatrix}
      +
      \begin{array}{c}
         m - \\
         m -
      \end{array}
   \right)
   \begin{tikzpicture}[baseline=(A.base), inner sep=1pt]
      \node (A) {$A$};
      \draw (A) -- ++(-.5, .1);
      \draw (A) -- ++(-.5, -.1);
   \end{tikzpicture}
   \\
   &&&=
   \mathbin{\begin{tikzpicture}[baseline=(A.base), inner sep=1pt]
      \node (n1) at (0,-.25) {$\tilde x$};
      \node (n2) at (0,.25) {$\tilde x$};
      \node at (-.4, 0) {$\Bigg[$};
      \node at (.4, 0) {$\Bigg]$};
      \node (A) at (.9, 0) {$A$};
      \draw (n1) -- (A);
      \draw (n2) -- (A);
   \end{tikzpicture}}
   +
   \begin{tikzpicture}[baseline=(A.base), inner sep=1pt]
      \node (A) {$A$};
      \node (m1) at (-.75, .2) {$m$};
      \node (m2) at (-.75, -.2) {$m$};
      \draw (m1) -- (A) -- (m2);
   \end{tikzpicture}
   \\
   &&&=
   \trace{M,A}2
   +
   \vecmatvec{.5em}{m}{A}{m}
   \\[.5em]
\end{walign}

The general outer-product form is obtained by the same ``pull out the mean'' step applied to two independent linear expressions.
We split $Ax+a = A\tilde x + (Am+a)$ and likewise for $Bx+b$; the centered-times-mean cross terms vanish in expectation.

\begin{walign}
   \tag{320}
   &\E[(Ax+a)(Bx+b)^T]
   &&
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{x} + a \\
      \vecmatvec{.5em}{B}{}{x} + b
   \end{bmatrix}
 \\*
   &= AMB^T + (Am+a)(Bm+b)^T
   &&=
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{\tilde x} \\
      \vecmatvec{.5em}{B}{}{\tilde x}
   \end{bmatrix}
   +
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{m} + a \\
      \vecmatvec{.5em}{B}{}{m} + b
   \end{bmatrix}
 \\*
   &
   &&=
   \vecmatvec{.5em}{}{A,M_2,B}{}
   +
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{m} + a \\
      \vecmatvec{.5em}{B}{}{m} + b
   \end{bmatrix}
\end{walign}

\noindent All the other outer-product identities are specializations of~(320) with $A$, $B$ set to $I$ or to each other, and $a$, $b$ optionally zero.
\begin{walign}
   \tag{321}
   \E[xx^T] &= M + mm^T
   &&
   \begin{bmatrix} x -\\x - \end{bmatrix}
   =
   \begin{tikzpicture}[baseline=(T.base), inner sep=1pt]
      \node (T) {$M$};
      \draw (T) -- ++(.5,.25);
      \draw (T) -- ++(.5,-.25);
   \end{tikzpicture}
   +
   \begin{matrix} m -\\ m - \end{matrix}
 \\[.4em]
   \tag{324}
   \E[(Ax)(Ax)^T] &= A(M+mm^T)A^T
   &&
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{x} \\ \vecmatvec{.5em}{A}{}{x}
   \end{bmatrix}
   =
   \vecmatvec{.5em}{}{A,M_2,A}{}
   +
   \renewcommand*{\arraystretch}{1.3}
   \begin{bmatrix}
      \vecmatvec{.5em}{A}{}{m} \\ \vecmatvec{.5em}{A}{}{m}
   \end{bmatrix}
 \\[.4em]
   \tag{325}
   \E[(x+a)(x+a)^T] &= M + (m+a)(m+a)^T
   &&
   \begin{bmatrix} x+a \\ x+a \end{bmatrix}
   =
   \begin{tikzpicture}[baseline=(T.base), inner sep=1pt]
      \node (T) {$M$};
      \draw (T) -- ++(.5,.25);
      \draw (T) -- ++(.5,-.25);
   \end{tikzpicture}
   +
   \begin{matrix} m+a\\ m+a \end{matrix}
\end{walign}

\noindent The remaining two outer-product identities are just~(321) contracted against a vector~$a$ (so one of the two free edges gets closed):
\begin{walign}
   \tag{322}
   \E[x\, a^T x] &= (M+mm^T)a
   &&
   \vecmatvec{.5em}{}{\E[xx^T]}{a}
   \;=\;
   \vecmatvec{.5em}{}{M_2}{a}
   \;+\;
   \vecmatvec{.5em}{}{m, m}{a}
 \\[.4em]
   \tag{323}
   \E[x^T a\, x^T] &= a^T(M+mm^T)
   && \text{(transpose of~(322))}
\end{walign}

\noindent Contracting the two free edges of~(320) against each other (i.e., taking the trace) turns the outer-product identities into scalar inner-product identities. The inner product $(Ax+a)^T(Bx+b)$ is just the two vectors joined on their free edges:
\begin{walign}
   \tag{326}
   &\E[(Ax+a)^T(Bx+b)]
   &&
   \vecmatvec{.5em}{(Ax+a)}{}{(Bx+b)}
 \\*
   &= \Tr(AMB^T) + (Am+a)^T(Bm+b)
   &&= \trace{A,M_2,B}{1.3}
 \\*
   &
   &&\phantom{{}={}}{}+ \vecmatvec{.5em}{(Am+a)}{}{(Bm+b)}
\end{walign}
\noindent The remaining trace identities specialize~(326) the same way:
\begin{walign}
   \tag{327}
   &\E[x^T x]
   &&
   \vecmatvec{.5em}{x}{}{x}
 \\*
   &= \Tr(M) + m^T m
   &&= \trace{M}{1.3}
   \;+\;
   \vecmatvec{.5em}{m}{}{m}
 \\[.4em]
   \tag{328}
   \E[x^T A x] &= \Tr(AM) + m^T A m
   && \text{(already shown as~(318))}
 \\[.4em]
   \tag{329}
   &\E[(Ax)^T(Ax)]
   &&
   \vecmatvec{.5em}{(Ax)}{}{(Ax)}
 \\*
   &= \Tr(AMA^T) + (Am)^T(Am)
   &&= \trace{A,M_2,A}{1.3}
   \;+\;
   \vecmatvec{.5em}{(Am)}{}{(Am)}
 \\[.4em]
   \tag{330}
   &\E[(x+a)^T(x+a)]
   &&
   \vecmatvec{.5em}{(x+a)}{}{(x+a)}
 \\*
   &= \Tr(M) + (m+a)^T(m+a)
   &&= \trace{M}{1.3}
   \;+\;
   \vecmatvec{.5em}{(m+a)}{}{(m+a)}
\end{walign}

\subsection{Cubic Forms}

The raw third moment $\E[x^{\otimes 3}]$ decomposes into central moments by the same ``pull out the mean'' expansion, $x = \tilde x + m$. Of the eight terms in the expansion of $(\tilde x + m)^{\otimes 3}$, the three terms containing exactly one centered factor vanish in expectation (since $\E[\tilde x] = 0$), leaving

\renewcommand*{\arraystretch}{1}
\begin{align*}
\begin{bmatrix}
   x - \\
   x - \\
   x -
\end{bmatrix}
&=
\begin{bmatrix}
   \tilde x - \\
   \tilde x - \\
   \tilde x -
\end{bmatrix}
+
3 \cdot \underbrace{\left(
   \begin{array}{l}
   \begin{bmatrix}
      \tilde x - \\
      \tilde x -
   \end{bmatrix}\\[.3em]
   \hphantom{[}\,m -
   \end{array}
\right)}_{\text{one of three positions}}
+
\begin{array}{c}
   m - \\
   m - \\
   m -
\end{array}
\\&=
\begin{tikzpicture}[baseline=(T.base), inner sep=1pt]
   \node (T) {$M_3$};
   \draw (T) -- ++(0,.4);
   \draw (T) -- ++(-.25,-.35);
   \draw (T) -- ++(.25,-.35);
\end{tikzpicture}
\;+\; 3
\begin{tikzpicture}[baseline=0pt, inner sep=1.5pt]
   % M node with two horizontal edges going left
   \node (M) at (0, -.3) {$M$};
   \draw (M.west) -- ++(-1.1, .2);
   \draw (M.west) -- ++(-1.1, -.2);
   % m node with one horizontal edge going left
   \node (m) at (0, .55) {$m$};
   \draw (m.west) -- ++(-1.1, 0);
   % vertical squiggle crossing all three free edges
   \draw [very thick, line width=1.5pt, decorate, decoration={snake, amplitude=0.45mm, segment length=2.2mm}]
         (-1.0, -.9) -- (-1.0, 1.0);
\end{tikzpicture}
\;+\;
\begin{array}{c}
   m - \\
   m - \\
   m -
\end{array}
\end{align*}
The squiggle denotes symmetrization over its edges (the average over all $3! = 6$ permutations of those legs, following the convention in §\ref{sec:symmetry}). Since $M$ is symmetric, permutations that only swap $M$'s two legs act trivially, so the average produces each of the $3$ distinct assignments of $m$ and $M$ to the legs with weight $\tfrac26 = \tfrac13$; the factor $3$ turns this average back into the sum of the three distinct terms.

For an affine expression $(Ax+a)(Bx+b)^T(Cx+c)$, the same pattern gives
\begin{align*}
   \E\!\left[(Ax+a)(Bx+b)^T(Cx+c)\right]
   &= A\cdot M_3\!\!\left(B^T, C\right)
   \\&\quad+ AMB^T(Cm+c) + AMC^T(Bm+b) + (Am+a)\,\Tr(BMC^T)
   \\&\quad+ (Am+a)(Bm+b)^T(Cm+c),
\end{align*}
where $M_3(B^T, C)$ denotes the order-1 tensor obtained by contracting two legs of $M_3$ with $B^T$ and~$C$ respectively, and then contracting the remaining free indices of $B$ and $C$ with each other; in components, $M_3(B^T, C)_i = (M_3)_{ijk} B_{lj} C_{lk}$.


\section{Cumulants}
\label{sec:cumulants}

Given a random vector $x\in\mathbb R^d$,
its $n$th Cumulant Tensor, $K_n\in\R^{d,\ldots,d}$ is defined by
\[
   \log \mathrm{E}(\mathrm{e}^{\langle t, x\rangle})
   = \sum_{n=1}^\infty \frac{1}{n!} \langle K_n, t^{\otimes n}\rangle
   =
   \begin{tikzpicture}[baseline=.5em, inner sep=1]
      \node (T) {$K_1$};
      \draw (T) -- ++(0,.5) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \frac{1}{2}
   \begin{tikzpicture}[baseline=(T.base), inner sep=1]
      \node (T) {$K_2$};
      \draw (T) -- ++(0,.5) node[fill=white] {$t$};
      \draw (T) -- ++(0,-.5) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \frac{1}{6}
   \begin{tikzpicture}[baseline=(T.base), inner sep=1]
      \node (T) {$K_3$};
      \draw (T) -- ++(0,.5) node[fill=white] {$t$};
      \draw (T) -- ++(-.4,-.4) node[fill=white] {$t$};
      \draw (T) -- ++(.4,-.4) node[fill=white] {$t$};
   \end{tikzpicture}
   +
   \dots
\]
The first couple of cumulants are similar to the central moments:
\[
   K_1 = m
   \quad\text{and}\quad
   K_2 = M
   \quad\text{and}\quad
   K_3 = M_3
   \quad\text{and}\quad
   K_4 = M_4
   - \begin{tikzpicture}[baseline=-.8em, inner sep=.5pt]
      \node (M1) {\scriptsize $M_2$};
      \node[below=.4em of M1] (M2) {\scriptsize $M_2$};
      \draw (M1) -- ++(.35,0);
      \draw (M2) -- ++(.35,0);
      \draw (M1) -- ++(-.35,0);
      \draw (M2) -- ++(-.35,0);
   \end{tikzpicture}
   - \begin{tikzpicture}[baseline=(M1.base), inner sep=.5pt]
      \node (M1) {\scriptsize $M_2$};
      \node[right=.05em of M1] (M2) {\scriptsize $M_2$};
      \draw (M1) -- ++(-.25,.25);
      \draw (M1) -- ++(-.25,-.25);
      \draw (M2) -- ++(.25,.25);
      \draw (M2) -- ++(.25,-.25);
   \end{tikzpicture}
   - \begin{tikzpicture}[baseline=(M1.base), inner sep=.5pt]
      \node (M1) {\scriptsize $M_2$};
      \node[right=.05em of M1, yshift=-2pt] (M2) {\scriptsize $M_2$};
      \draw (M1) -- ++(-.25,.25);
      \draw (M1) -- ++(+.7,-.25);
      \draw (M2) -- ++(-.7,-.25);
      \draw (M2) -- ++(.25,.25);
   \end{tikzpicture}
   .
\]
They have the nice property (which is easy to see from the definition) that they are additive for independent random variables:
$
   K_n(x + y) = K_n(x) + K_n(y).
$
This generalizes the standard property that the variance of the sum of independent random variables is the sum of the variances.

We can write the expectations of $x^{\otimes n}$ in terms of the cumulants:

\[
   \begin{bmatrix}
      \begin{tikzpicture}[
         baseline=(T.base),
         every node/.style={inner sep=1pt, circle, draw=none, fill=white},
       ]
       \draw (30:.2) node[label] {$x$} -- (30:.6);
       \draw (150:.2) node[label] {$x$} -- (150:.6);
       \draw (270:.2) node[label] {$x$} -- (270:.6);
      \end{tikzpicture}
   \end{bmatrix}
   =
   \begin{tikzpicture}[
      baseline=-1em,
     every node/.style={inner sep=1pt, circle, draw, fill=white, thin},
     label/.style={draw=none, fill=none, text=black, font=\scriptsize},
     declare function={
       xgap=1.8;  % Horizontal spacing between diagrams
     }]
     \drawPartitionAtAngle[k]{(-1*xgap,0)}{3}{}{{{1,2,3}}}
     \drawPartitionAtAngle[k]{(0,0)}{3}{$+$}{{{1,2},{3}}}
     \drawPartitionAtAngle[k]{(xgap,0)}{3}{$+$}{{{1,3},{2}}}
     \drawPartitionAtAngle[k]{(xgap*2,0)}{3}{$+$}{{{1},{2,3}}}
     \drawPartitionAtAngle[k]{(xgap*3,0)}{3}{$+$}{{{1},{2},{3}}}
   \end{tikzpicture}
   .
\]
In general the sum is over all the partitions of the set $\{1,\ldots,n\}$.

If the entries of $x$ are \emph{independent} and identically distributed with scalar cumulants $\kappa_1, \kappa_2, \kappa_3, \kappa_4 \in \R$, the off-diagonals of the cumulant tensors $K_1, K_2, \dots$ all vanish, and the partition sum reduces to a scalar weight times a ``copy tensor'' (diagonal) for each block. For $n=4$:
\[
\begin{bmatrix}
   \begin{tikzpicture}[
      baseline=(T.base),
      every node/.style={inner sep=1pt, circle, draw=none, fill=white},

\section{Gaussian Moments}

For a Gaussian vector $x \sim \mathcal{N}(m, M)$, all cumulants beyond the second vanish: $K_1 = m$, $K_2 = M$, and $K_n = 0$ for $n \ge 3$. The moment-cumulant formula from the Cumulants section then collapses to a sum over partitions of $\{1,\dots,n\}$ into singletons (each carrying a factor of $m$) and pairs (each carrying a factor of $M$)~--- \emph{Isserlis' theorem}, also known as \emph{Wick's theorem}:
\[
   \left[
      \begin{tikzpicture}[baseline=(c.base), inner sep=1pt, every node/.style={inner sep=1pt}]
         \node (c) at (0, 0) {$x$};
         \node[right=.3em of c] (dots) {$\cdots$};
         \node[right=.3em of dots] (cn) {$x$};
         \draw (c) -- ++(0, .35);
         \draw (cn) -- ++(0, .35);
         \node[above=.3em of dots] {\scriptsize $n$ copies};
      \end{tikzpicture}
   \right]
   = \sum_{\pi \in \mathcal{P}(n)} \prod_{B \in \pi}
   \begin{cases} m & |B| = 1 \\ M & |B| = 2 \\ 0 & |B| \ge 3 \end{cases}.
\]
For centered Gaussians ($m = 0$) only the pairings survive, giving the familiar identity
\[
   E[x^{\otimes 2k}] = \sum_{\pi \text{ pairing}} \bigotimes_{(i,j) \in \pi} M,
\]
a sum with $(2k-1)!!$ terms ($3$ for $k=2$, $15$ for $k=3$).

\subsection{Quadratic and Quartic Forms}

The quadratic-form identities of the Quadratic Forms section hold for arbitrary distributions, so they apply to Gaussians without modification --- see (318) and (321). What \emph{is} Gaussian-specific is the variance of a quadratic form, where, for symmetric $A$, Wick pairing leaves only two surviving terms after cancelling~$\E[x^T A x]^2$:
\begin{align*}
   \mathrm{Var}[x^T A x]
   &= 2\,\Tr(AMAM) + 4\, m^T A M A m \tag{379}
   \\ &= 2\;\trace{A,M,A,M}{1} \;+\; 4\;\vecmatvec{.5em}{m}{A,M,A}{m}\,.
\end{align*}
Specializing to $M = \sigma^2 I$ recovers the Matrix Cookbook identity
\begin{equation}
   \mathrm{Var}[x^T A x] = 2\sigma^4 \Tr(A^2) + 4\sigma^2 m^T A^2 m. \tag{381}
\end{equation}
For two symmetric forms with the same Gaussian,
\begin{equation}
   \mathrm{Cov}(x^T A x,\, x^T B x) = 2\,\Tr(A M B M) + 4\, m^T A M B m. \tag{382}
\end{equation}
(Our version generalizes the Matrix Cookbook's (382), which assumes $x \sim \mathcal{N}(0, \sigma^2 I)$ and hence reads $2\sigma^4 \Tr(AB)$.)

More general fourth moments follow the same Wick-pairing recipe: given any four linear expressions $\ell_1(x), \dots, \ell_4(x)$, we have
\[
   \E[\ell_1 \ell_2 \ell_3 \ell_4]
   = \sum_{\text{3 pairings}} \E[\ell_i \ell_j]\, \E[\ell_k \ell_l] + \text{mean-only terms}.
\]
For example,
\begin{align*}
   \E[a^T x \, b^T x \, c^T x \, d^T x]
   &= (a^T M b)(c^T M d)
   + (a^T M c)(b^T M d)
   + (a^T M d)(b^T M c)
   \\ &\quad + \text{mean-dependent terms},
\end{align*}
which the reader can unfold by writing $\ell_i = a_i^T\tilde x + a_i^T m$ and collecting terms.

\subsection{Mixture of Gaussians}
For a mixture
\(
   x \sim \sum_k \rho_k\, \mathcal{N}(m_k, M_k),
\)
linearity of expectation gives
\begin{align*}
   \tag{384}
   \E[x] &= \sum_k \rho_k\, m_k
   \\
   \tag{385}
   \mathrm{Var}[x] &= \sum_k \rho_k (M_k + m_k m_k^T) - \left(\sum_k \rho_k m_k\right)\!\left(\sum_k \rho_k m_k\right)^{\!T}.
\end{align*}
Higher moments follow the same pattern --- each component contributes its own Wick-pairing sum, weighted by $\rho_k$.

\subsection{Derivatives of the Gaussian Log-Likelihood}

For $x \sim \mathcal{N}(m, M)$ with density
\[
   p(x;\, m, M) = (2\pi)^{-n/2} \det(M)^{-1/2} \exp\!\left(-\tfrac{1}{2}(x-m)^T M^{-1} (x-m)\right),
\]
the log-likelihood score functions are
\begin{align*}
   \frac{\partial \log p}{\partial m} &= M^{-1}(x - m),
   \\
   \frac{\partial \log p}{\partial M} &= \tfrac{1}{2}\, M^{-1}\!\left[(x-m)(x-m)^T - M\right]\! M^{-1}.
\end{align*}
Both have zero expectation under~$p$, as scores must. Setting the sample scores to zero, the MLE estimators from $N$ i.i.d.\ samples $x_1, \dots, x_N$ are
\[
   \hat m = \frac{1}{N} \sum_i x_i,
   \qquad
   \hat M = \frac{1}{N} \sum_i (x_i - \hat m)(x_i - \hat m)^T.
\]
The expected Fisher information follows from a second differentiation and Stein's lemma; it equals the covariance of the score, $\E\!\left[\nabla \log p \, (\nabla \log p)^T\right]$, which we can evaluate with the tensor Wick theorem above.

\section{Free Probability for Random Matrices}

When we move from scalar random variables to matrices, two things change: matrix products don't commute, and the dimension $N$ becomes a parameter in its own right. The payoff: as $N \to \infty$, many matrix ensembles have limiting spectra that are governed by a clean analogue of the cumulants of \S\ref{sec:cumulants}~-- \emph{free cumulants}~-- where the sum-over-partitions gets restricted to \emph{non-crossing} partitions. This single adjustment captures both the Wigner semicircle (free Gaussian) and the Marchenko--Pastur (free $\chi^2$) distributions within the same formal framework.

   m_n(X) = \lim_{N \to \infty} \frac{1}{N}\, \E[\Tr(X^n)],
   \qquad
   m_n(X) = \sum_{\pi \in \mathrm{NC}(n)} \prod_{B \in \pi} \kappa_{|B|}^{\mathrm{free}}(X).
\]
The second equation is Speicher's moment-cumulant formula~-- the same sum-over-partitions as in \S\ref{sec:cumulants}, but restricted to non-crossing ones. Writing $\kappa_k$ for $\kappa_k^{\mathrm{free}}$ and expanding for small $n$:
\begin{align*}
   m_1 &= \kappa_1, \\
   m_2 &= \kappa_2 + \kappa_1^2, \\
   m_3 &= \kappa_3 + 3\kappa_2\kappa_1 + \kappa_1^3, \\
   m_4 &= \kappa_4 + 4\kappa_3\kappa_1 + 2\kappa_2^{\,2} + 6\kappa_2\kappa_1^{\,2} + \kappa_1^4.
\end{align*}
The coefficient $2$ on $\kappa_2^{\,2}$ (not $3$ as in the classical case) is the key difference: the pair-pair pattern has three set-partitions $\{1,2\}\{3,4\},\{1,3\}\{2,4\},\{1,4\}\{2,3\}$, but the middle one crosses and drops out.

\subsection{R-Transform and Free Convolution}

The \emph{R-transform} of $X$ is the generating function of free cumulants,
\[
   R_X(z) = \sum_{n \ge 1} \kappa_n^{\mathrm{free}}(X)\, z^{n-1}.
\]
The crucial property: if $X$ and $Y$ are \emph{freely independent} (the non-commutative analogue of ``classical independence'' for large random matrices~-- essentially, their eigenvector bases are asymptotically generic with respect to each other), then
\[
   R_{X+Y}(z) = R_X(z) + R_Y(z).
\]
This is the direct free analogue of the classical statement that cumulants add under independence. For free multiplicative convolution (the distribution of $X^{1/2} Y X^{1/2}$ when $X, Y$ are free positive), the corresponding transform is the $S$-transform and is multiplicative.

\subsection{Two Canonical Distributions}

\paragraph{Wigner semicircle} (free analogue of the standard Gaussian). Density
\(
   \rho(x) = \frac{1}{2\pi}\sqrt{4 - x^2}
\)
on $[-2, 2]$. All free cumulants vanish except $\kappa_2 = 1$, so $R(z) = z$. It arises as the $N\to\infty$ limit of the spectrum of an $N\times N$ Gaussian symmetric matrix (GOE/GUE) with entries of variance $1/N$. Just as the classical central limit theorem makes sums of i.i.d.\ scalars Gaussian, the \emph{free} CLT makes sums of free self-adjoint matrices semicircular.

\paragraph{Marchenko--Pastur} (free analogue of the scaled $\chi^2$). For the Wishart matrix $W = \frac{1}{N} X X^T$ with $X \in \R^{N \times cN}$ having i.i.d.\ zero-mean unit-variance entries, the limiting spectrum is $\mathrm{MP}(c)$ with $R$-transform
\(
   R(z) = \frac{c}{1 - z}.
\)

\subsection{A Note on Higher-Rank Tensors}

For random tensors of rank $\ge 3$ a partial analogue of free probability exists; see e.g.\ Gurau's \emph{Random Tensors} (Oxford, 2017). The relevant combinatorial object is \emph{colored non-crossing partitions}, counted by the Fuss--Catalan numbers $C_n^{(D)} = \frac{1}{(D-1)n + 1}\binom{Dn}{n}$ (so $C_n^{(2)} = C_n$). The analogue of the non-crossing-pairing dominance for matrices is the dominance of \emph{melonic} diagrams in the large-$N$ limit. The theory is genuinely richer than the matrix case and we do not develop it further here.


\section{Exercises}
\begin{exercise}
   Assume the coordinates of $x$ are independent, and let $v_3$ denote the vector of their third central moments, $(v_3)_i = \E[(x_i - m_i)^3]$. Show that
   \begin{align*}
   \mathrm{E}[(Ax + a)(Ax + a)^T (Ax + a)] =
   A\,\diag(A^T A)\, v_3 \\
   + [2 AMA^T + (Am + a)(Am + a)^T] (Am + a) \\
   + \mathrm{Tr}(AMA^T)(Am + a)
   \end{align*}
   Assuming instead that $x$ is Gaussian, show that
   \begin{align*}
   \mathrm{E}[(Ax + a) b^T (Cx + c)(Dx + d)^T] =
   (Am + a) b^T (CMD^T + (Cm + c)(Dm + d)^T) \\
   + (AMC^T + (Am + a)(Cm + c)^T) b(Dm + d)^T \\
   + b^T (Cm + c)(AMD^T - (Am + a)(Dm + d)^T)
   \end{align*}
\end{exercise}

\end{exercise}
